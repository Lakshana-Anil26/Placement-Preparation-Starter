<script>

const chatbot=document.getElementById("chatbot");
const chatBody=document.getElementById("chatBody");

function toggleChat(){
    chatbot.classList.toggle("active");
}

function handleKey(e){
    if(e.key==="Enter") sendMessage();
}

function sendMessage(){
    const input=document.getElementById("userInput").value.trim();
    if(input==="") return;

    appendMessage("You",input,"user-msg");
    document.getElementById("userInput").value="";

    showTyping();

    setTimeout(()=>{
        removeTyping();
        const reply=generateReply(input);
        appendMessage("AI",reply,"bot-msg");
        saveChat();
    },1200);
}

function appendMessage(sender,text,className){
    chatBody.innerHTML+=`<div class="${className}"><b>${sender}:</b> ${text}</div>`;
    chatBody.scrollTop=chatBody.scrollHeight;
}

function showTyping(){
    chatBody.innerHTML+=`<div class="typing" id="typing">AI is typing...</div>`;
    chatBody.scrollTop=chatBody.scrollHeight;
}

function removeTyping(){
    const typing=document.getElementById("typing");
    if(typing) typing.remove();
}

function generateReply(input){
    input=input.toLowerCase();
    if(input.includes("roadmap")) return "Follow the 4-year roadmap step by step.";
    if(input.includes("coding")) return "Practice DSA daily using Coding Assistant.";
    if(input.includes("resume")) return "Resume Builder helps create ATS-friendly resumes.";
    if(input.includes("mock")) return "AI Mock Interviews simulate real company interviews.";
    if(input.includes("timetable")) return "Weekly Timetable helps manage your study plan.";
    return "Stay consistent and focus on DSA, projects and communication skills.";
}

function saveChat(){
    localStorage.setItem("chatHistory",chatBody.innerHTML);
}

function loadChat(){
    const history=localStorage.getItem("chatHistory");
    if(history) chatBody.innerHTML=history;
    else appendMessage("AI","Hello! Ask me anything about placements 😊","bot-msg");
}

function clearChat(){
    localStorage.removeItem("chatHistory");
    chatBody.innerHTML="";
    appendMessage("AI","Chat cleared. How can I help you?","bot-msg");
}

loadChat();

</script>
